source ../botany-linux/.env

sudo \
    HOOK_post_rootfs=$(realpath ../botany-linux/iso_files/configure_iso_anaconda.sh) \
    HOOK_pre_initramfs=$(realpath ../botany-linux/iso_files/configure_iso_anaconda-pre.sh) \
    REGISTRY_AUTH_FILE=$(realpath ../botany-linux/auth_puller.json) \
just build "${IMAGE_REGISTRY}/${IMAGE_NAME}:prod" "1" "$(realpath ../botany-linux/system_files/usr/share/ublue-os/system-flatpaks.list)"

sudo chown $(id -u):$(id -g) ./output.iso && mv ./output.iso "./botany-linux_$(printf '%(%Y-%m-%d)T' -1).iso"

sudo cp -f --reflink=always "./botany-linux_$(printf '%(%Y-%m-%d)T' -1).iso" "/var/lib/libvirt/images/botany-linux.iso"

qemu-img create -f qcow2 botany-linux.qcow2 40G
